    This file is the Strapi rich text renderer (NOT the layout block-renderer at `custom/layouts/block-renderer.tsx`). It imports `BlocksContent` and `InlineNode` which were removed. Per the verification notes, this needs replacement with a Portable Text stub (full renderer is Phase 3, RT-01).

    Replace the entire file with a Portable Text stub renderer:
    ```typescript
    import React from "react";
    import type { TPortableTextBlock } from "@/types";

    interface BlockRendererProps {
      content: TPortableTextBlock[];
    }

    /**
     * Portable Text stub renderer.
     * Extracts plain text from Portable Text blocks.
     * Full styled renderer with @portabletext/react built in Phase 3 (RT-01).
     */
    export function BlockRenderer({ content }: BlockRendererProps) {
      if (!content || !Array.isArray(content)) {
        return null;
      }

      return (
        <div className="prose max-w-none">
          {/* Portable Text renderer — Phase 3 */}
          {content.map((block) => (
            <p key={block._key} className="mb-4 mx-auto indent-3.5 text-base font-normal px-2 max-w-3xl">
              {block.children?.map((child) => child.text).join("")}
            </p>
          ))}
        </div>
      );
    }
    ```

    This preserves the `BlockRenderer` export name and `content` prop interface so `cta-feature.tsx` (if it still imports it) continues to work. The stub renders plain text from Portable Text blocks — same pattern used in `image-with-text.tsx`, `info-section.tsx`, and `cta-features-section.tsx` from Plan 03.
